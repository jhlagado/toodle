export const logout = createAction(LOGOUT);
 
export const showLogoutModal = () => showBasicModal("Logout", "Do you want to logout?", [
    { id: 'ok', title: 'OK', onClick: logout },
    { id: 'cancel', title: 'Cancel' },
]);
